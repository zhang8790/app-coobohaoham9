// 订单创建时
import { runOrderRiskCheck } from '@/utils/risk-control';

async function createOrder(orderData) {
  // 1. 风控检测
  const riskResult = await runOrderRiskCheck(supabase, {
    userId: orderData.userId,
    staffId: orderData.staffId,
    storeId: orderData.storeId,
    referrerId: orderData.referrerId,
    orderAmount: orderData.amount,
    orderId: orderData.id,
  });
  
  if (!riskResult.passed) {
    // 高风险：冻结佣金
    if (riskResult.shouldFreeze) {
      orderData.commission_frozen = true;
      orderData.freeze_reason = riskResult.risks.map(r => r.description).join('; ');
    }
  }
