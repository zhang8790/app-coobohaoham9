    ```sql
    DROP POLICY IF EXISTS rls81_profiles_downline_read ON public.profiles;
    CREATE OR REPLACE FUNCTION public.fn_my_l1_referral_ids(p_uid uuid)
    RETURNS uuid[] LANGUAGE sql SECURITY DEFINER SET search_path=public AS $$
      SELECT COALESCE(array_agg(id), '{}'::uuid[]) FROM public.profiles WHERE referrer_id=p_uid
    $$;
    CREATE POLICY rls81_profiles_downline_read ON profiles FOR SELECT TO authenticated
    USING (referrer_id = auth.uid() OR referrer_id = ANY(public.fn_my_l1_referral_ids(auth.uid())));
    ```
